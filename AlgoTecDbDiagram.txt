Table Users as U {
  Id bigint [pk, increment]
  Name varchar (500)
  Surname varchar (500)
  Phone varchar (500)
  Email varchar (500)
  DateOfBirth timestamp
 }
Table Space as S {
  Id bigint [pk, increment] 
  TypeOfSpaceId int
  SpaceProperty json
 }
Table Contracts as C {
  Id bigint [pk, increment]
  OwnerUserId bigint
  TenantUserId bigint
  SpaceId bigint
}
Table SpaceAddresses as SA {
  Id bigint [pk, increment] 
  CountryId int
  CityId bigint
  StreetId bigint
  PostIndexId bigint
  Number int
  SpaceId bigint
 } 
Table TypeOfSpace as TO {
  Id int [pk, increment] 
  Type varchar (500)
 }
Table UserAddresses as UA {
  Id bigint [pk, increment] 
  CountryId int
  CityId bigint
  StreetId bigint
  PostIndexId bigint
  Number int
  Pavilion varchar(500)
  ApartmentNumber varchar(500)
  UserId bigint
 } 
Table Countries as CO {
  Id int [pk, increment] 
  Name varchar (500)
 } 
Table Cities as CI {
  Id bigint [pk, increment]
  CountryId int
  Name varchar (500)
 }
Table Streets as ST {
  Id bigint [pk, increment] 
  CityId bigint
  Name varchar (500)
 }
Table PostIndexes as PI {
  Id bigint [pk, increment] 
  CityId bigint
  Name varchar (500)
 }
